#!/bin/bash
# Verification script to check if camera overlays are properly set up in the RPi5 minimal image
# This script can be run on the Pi after first boot to verify the setup

echo "=== RPi5 Camera Overlay Verification Guide ==="
echo ""
echo "Run these commands on the Pi after booting with the new image:"
echo ""
echo "1. Verify overlays are in /boot/overlays/ (should be copied automatically at first boot):"
echo "   ls -la /boot/overlays/ | grep imx477"
echo ""
echo "2. Verify kernel modules are loaded:"
echo "   lsmod | grep -E '(imx477|bcm2835_unicam|rp1_cfe|videobuf2)'"
echo ""
echo "3. Verify device tree overlay is loaded in config.txt:"
echo "   grep -E 'dtoverlay|camera_auto_detect' /boot/firmware/config.txt"
echo ""
echo "4. Check kernel logs for camera-related messages:"
echo "   dmesg | grep -i 'imx477\|cfe\|camera' | tail -20"
echo ""
echo "5. List available cameras (this is the real test):"
echo "   libcamera-hello --list-cameras"
echo ""
echo "6. If step 5 fails, check these diagnostic outputs:"
echo "   - SMI (Senor Module Interface) status: vcgencmd get_camera"
echo "   - I2C devices (camera should show up): i2cdetect -y 0"
echo "   - Kernel module load order: journalctl | grep -i 'imx477\|camera'"
echo ""
echo "Expected successful output for libcamera-hello --list-cameras:"
echo "   Available cameras"
echo "   0 : imx477 [4608x3456]"
echo ""
echo "If you see 'no cameras available', check:"
echo "   1. Are overlays in /boot/overlays/? (if not, init script failed)"
echo "   2. Is config.txt correct? (grep dtoverlay)"
echo "   3. Is CSI connector working? (vcgencmd status)"
echo ""
